＊main_process.py*
def run_pipeline(input_file):

    df = pd.read_excel(input_file)

    # データ加工
    df = preprocess.compress(df)
    df = preprocess.convert_numeric(df)

    # 名簿修正
    df = employee.fix_employee_id(df)

    # ソート
    df = sort_data.sort_main(df)

    # 組織DB
    df = org_db.merge_org(df)

    # 基本チェック
    basic_result = basic_check.run(df)

    # 打忘れ
    forgot_result = forgot_count.run(df)

    # 公休日労働
    holiday_result = holiday_work.run(df)

    # 公休日労働回数
    holiday_count_result = holiday_count.run(holiday_result)

    # 特休
    special_result = special_leave.run(df)

    # 出力
    export.run(
        basic_result,
        forgot_result,
        holiday_result,
        holiday_count_result,
        special_result
    )

import pandas as pd

def export_all(
    basic_result,
    forgot_result,
    holiday_result,
    holiday_count_result,
    special_result
):

    with pd.ExcelWriter("output/result.xlsx", engine="openpyxl") as writer:
        basic_result.to_excel(writer, sheet_name="基本チェック", index=False)
        forgot_result.to_excel(writer, sheet_name="打忘回数", index=False)
        holiday_result.to_excel(writer, sheet_name="公休日労働実績", index=False)
        holiday_count_result.to_excel(writer, sheet_name="公休日労働回数", index=False)
        special_result.to_excel(writer, sheet_name="特休公休", index=False)
